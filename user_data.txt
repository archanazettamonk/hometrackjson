#!/bin/bash
mkdir /var/www
chown -R ec2-user:ec2-user /var/www
chmod g+s /var/www
chmod g+w -R /var/www

sudo yum -y install git

#node
cd /home/ec2-user
mkdir nvm
chown ec2-user:ec2-user nvm
chmod 777 nvm
export NVM_DIR="/home/ec2-user/nvm"
curl -o- https://raw.githubusercontent.com/creationix/nvm/v0.32.0/install.sh | bash
[ -s "$NVM_DIR/nvm.sh" ] && . "$NVM_DIR/nvm.sh"
nvm install 4.2.1

cd /var/www/
git clone https://github.com/archanazettamonk/hometrackjson.git
mv hometrackjson html
cd html
npm install
nodemon app.js
